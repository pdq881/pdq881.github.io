        balances[_to]        = balances[_to].add(_value);
        emit Transfer(msg.sender, _to, _value);

        return balances[_owner];

        balances[_to]          = balances[_to].add(_value);

        return true;

        allowed[msg.sender][_spender] = _value;
        emit Approval(msg.sender, _spender, _value);
        return true;
    }
